import math

x = float(input('Введите x = '))
y = float(input('Введите y = '))
if (math.sqrt(x**2+y**2)<=2) and (x<0 and y>=-(x)**2+2) or (math.sqrt(x**2+y**2)<=2) and (x>0 and y>=-(x)**2+2):
print('Принадлежит')
else:
print('Не принадлежит')
