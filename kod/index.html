import random
n = int(input('Введите количество строк = '))
m = int(input('Введите количество стобцов = '))
a = [[random.randrange(0,10) for j in range(m)] for i in range(n)]
nm = []
print('Дана матрица: ')
for row in a:
print(' '.join([str(elem) for elem in row]))
def am(nm):
nm = [sum(row[i] for row in a) for i in range(len(a[0]))]
print('Норма матрицы ||Am|| = ', nm)
def ai(nm):
nm = [sum(row) for row in a]
print('Норма матрицы ||Ai|| = ', nm)
am(nm)
ai(nm)
